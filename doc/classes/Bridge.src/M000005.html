  def self.publish_service svc, fun
    if svc == 'system'
      Util::err('Invalid service name: ' + svc)
    else
      Core::command(:JOINWORKERPOOL,
                    { :name     => svc,
                      :callback => Core::store(fun.hash, CallbackRef(fun))
                    })
    end
    Core::store(name, svc)
  end
