def receive_data data
  left = @len - @pos
  if data.length >= left
    @buffer << data.slice(0, left)
    @callback.call @buffer
    receive_data data.slice(left..-1) unless data.nil?
  else
    @buffer << data
    @pos = @pos + data.length
  end
end

def start
  # Read header bytes
  read 4 do |data|
    # Read body bytes
    read data.unpack('N')[0] do |data|
      # Call message handler
      @connection.onmessage({:data => data}, self)
      # Await next message
      start
    end
  end
end
