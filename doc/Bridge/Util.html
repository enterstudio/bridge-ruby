def self.inflate obj
  o = {}
  obj.each do |k, v|
    if v.respond_to? :keys
      o[k] = inflate v
    else
      o[k] = (v.respond_to?(:call) && !v.respond_to?(:path)) ? cb(v) : v
    end
  end
  o
end
