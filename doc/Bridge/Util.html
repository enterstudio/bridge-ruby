def self.bloat v
  if v.is_a?(Module) || v.is_a?(Bridge::Service)
    local_ref(v)
  elsif v.respond_to?(:call) && !v.is_a?(Ref)
    cb(v)
  elsif v.is_a?(Hash) || v.is_a?(Array)
    inflate v
  else
    v
  end
end

def self.deflate obj
  if obj.is_a?(Array)
    obj.map do |v|
      deflate v
    end
  elsif obj.is_a? Hash
    if obj.has_key? 'ref'
      Core::lookup obj['ref']
    else
      o = {}
      obj.each do |k, v|
        o[k] = deflate v
      end
      o
    end
  else
    obj
  end
end

def self.has_keys? obj, *keys
  keys.each do |k|
    if !obj.has_key?(k) && !obj.has_key?(k.to_sym)
      return false
    end
  end
  true
end

def self.inflate obj
  if obj.is_a?(Array)
    obj.map do |v|
      Util::bloat v
    end
  else
    o = {}
    obj.each do |k, v|
      o[k] = Util::bloat v
    end
    o
  end
end
